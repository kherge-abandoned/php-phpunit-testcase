PHPUnit TestCase

A PHPUnit test case class with additional functionality.

Usage

use Herrera\PHPUnit\TestCase;

class MyTest extends TestCase
{
    public function testCallProtected()
    {
        $object = new MyClass();

        $this->callMethod($object, 'protectedMethod', array('args'));
    }
}
